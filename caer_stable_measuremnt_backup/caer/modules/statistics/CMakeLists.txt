IF (NOT ENABLE_STATISTICS)
	SET(ENABLE_STATISTICS 0 CACHE BOOL "Enable the console statistics module")
ENDIF()

IF (ENABLE_STATISTICS)
	SET(CAER_COMPILE_DEFINITIONS ${CAER_COMPILE_DEFINITIONS} -DENABLE_STATISTICS=1 PARENT_SCOPE)

	SET(CAER_STAT_FILES modules/statistics/statistics.c)

	SET(CAER_C_SRC_FILES ${CAER_C_SRC_FILES} ${CAER_STAT_FILES} PARENT_SCOPE)
ENDIF()
